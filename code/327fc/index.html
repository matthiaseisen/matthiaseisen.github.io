import random, reportlab.platypus as platypus, reportlab.lib.colors as colors, reportlab.lib.units as units
platypus.BaseDocTemplate('1000-bingo-cards.pdf', pageTemplates=[platypus.PageTemplate(frames=[platypus.Frame(0, -5 * units.cm, 21 * units.cm, 29.7 *units.cm)])]).build([e for p in zip([platypus.Table([['B', 'I', 'N', 'G', 'O']] + [row if row_index != 2 else [v if i != 2 else None for i, v in enumerate(row)] for row_index, row in enumerate([list(row) for row in zip(*[sorted(random.sample(range(1, 76)[i * 15: (i + 1) * 15], 5)) for i in range(5)])])], 5 * [2.5 * units.cm], 6 * [2.5 * units.cm], style=platypus.TableStyle([('FONTSIZE', (0, 0), (-1, -1), 28), ('LEADING', (0, 0), (-1, -1), 28), ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'), ('INNERGRID', (0, 0), (-1, -1), 0.25, colors.black), ('BOX', (0, 0), (-1, 0), 2.0, colors.black), ('BOX', (0, 1), (-1, -1), 2.0, colors.black), ('ALIGN', (0, 0), (-1, -1), 'CENTER'), ('VALIGN', (0, 0), (-1, -1), 'MIDDLE'), ('BACKGROUND', (2, 3), (2, 3), colors.lightgrey), ('BACKGROUND', (0, 0), (-1, 0), colors.lightgrey)])) for i in range(1000)], [platypus.PageBreak()] * 1000) for e in p])
